actor AmongUsMegaHealth : CustomInventory
{
  +COUNTITEM
  +INVENTORY.AUTOACTIVATE
  +INVENTORY.ALWAYSPICKUP
  Inventory.MaxAmount 0
  Inventory.PickupMessage "Mega Health!"
  States
  {
    Spawn:
      TNT1 A -1
      Stop
    Pickup:
      TNT1 A 0 A_GiveInventory("MegasphereHealth", 1)
      Stop
  }
}
