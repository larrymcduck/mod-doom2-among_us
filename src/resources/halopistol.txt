actor  AmongUsM6d : Weapon 10013
{
	Tag "M6D Pistol"
	Scale 0.05
	weapon.ammouse 1
	weapon.ammotype "clip"
	+weapon.noautofire
	attacksound "weapons/m6dfire"
	Decal BulletChip
	States
	{
		Spawn:
			M6DP A -1
			Stop
		Ready:
			M6DP B 1 A_weaponready
			Loop
		Select:
			M6DP B 1 A_raise
			Loop
		Deselect:
			M6DP B 1 A_lower
			Loop
		Fire:
			M6DP C 2 A_FireBullets(4, 3, 1, 6, "BulletPuff")
			M6DP D 3
			M6DP E 5 A_WeaponReady
			Goto Ready
		AltFire:
			M6DP B 0 A_JumpIfInventory("M6dZoom", 1, "ZoomOut")
		ZoomIn:
			M6DP B 0 A_GiveInventory("M6dZoom", 1)
			M6DP B 2 A_ZoomFactor(2.0)
			Goto Ready
		ZoomOut:
			M6DP B 0 A_TakeInventory("M6dZoom", 1)
			M6DP B 2 A_ZoomFactor(1.0)
			Goto Ready
			
	}
}

ACTOR M6dZoom : Inventory { Inventory.MaxAmount 1 }
